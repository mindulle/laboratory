@JSName('copyTo')
Future<Entry> copyTo(DirectoryEntry parent, {String? name}) {
  var completer = new Completer<Entry>();
  _copyTo(parent, name, (value) {
    completer.complete(value);
  }, (error) {
    completer.completeError(error);
  });
  return completer.future;
}
